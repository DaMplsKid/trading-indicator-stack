// Clamp function
f_clamp(val, minVal, maxVal) =>
    math.max(minVal, math.min(maxVal, val))

// Sigmoid function
f_sigmoid(x) =>
    1.0 / (1.0 + math.exp(-x))

// IoU for 1D intervals (bands)
f_iou1d(a1, a2, b1, b2) =>
    intersectStart = math.max(a1, b1)
    intersectEnd = math.min(a2, b2)
    intersectLen = math.max(0, intersectEnd - intersectStart)
    unionLen = a2 - a1 + b2 - b1 - intersectLen
    if unionLen <= 0
        0.0
    else
        intersectLen / unionLen

// Exponential Moving Average helper
f_ema_update(prevEma, newValue, alpha = 0.2) =>
    na(prevEma) ? newValue : alpha * newValue + (1 - alpha) * prevEma

// Mini-convolution function (causal)
f_conv(src, weights) =>
    result = 0.0
    weightsSize = array.size(weights)
    if weightsSize > 0
        maxIdx = math.min(weightsSize - 1, 5) // Limit convolution window for performance
        for i = 0 to maxIdx by 1
            if bar_index >= i
                result := result + nz(src[i], 0) * array.get(weights, i)
                result
    result

// Manual ADX calculation (simplified)
f_adx(len = 14) =>
    up = ta.change(high)
    down = -ta.change(low)
    trur = ta.rma(ta.tr, len)
    plus = fixnan(100 * ta.rma(up > down and up > 0 ? up : 0, len) / trur)
    minus = fixnan(100 * ta.rma(down > up and down > 0 ? down : 0, len) / trur)
    sum = plus + minus
    adx = 100 * ta.rma(math.abs(plus - minus) / (sum == 0 ? 1 : sum), len)
    adx

// Volume Z-Score by time-of-day
var array<float> volMeanByHour = array.new<float>(24, 0.0)
var array<float> volStdByHour = array.new<float>(24, 1.0)
var array<int> volCountByHour = array.new<int>(24, 0)

// Update volume statistics by hour

    currentMean = array.get(volMeanByHour, currentHour)
    currentStd = array.get(volStdByHour, currentHour)
    currentCount = array.get(volCountByHour, currentHour)

    newCount = currentCount + 1
    newMean = currentMean + (currentVol - currentMean) / newCount
    newVariance = currentCount > 0 ? ((currentCount - 1) * math.pow(currentStd, 2) + math.pow(currentVol - newMean, 2)) / newCount : math.pow(currentVol - newMean, 2)
    newStd = math.max(currentVol * 0.1, math.sqrt(newVariance)) // Floor to avoid division by zero

    array.set(volMeanByHour, currentHour, newMean)
    array.set(volStdByHour, currentHour, newStd)
    array.set(volCountByHour, currentHour, newCount)

volZ = (currentVol - array.get(volMeanByHour, currentHour)) / array.get(volStdByHour, currentHour)

var map<int, float> binWeightedCounts = map.new<int, float>()
var map<int, float> binPriceSums = map.new<int, float>()
var map<int, int> binPriceCounts = map.new<int, int>()
var map<int, int> binPolarities = map.new<int, int>()
var int lastProcessedPivotIndex = 0

// Optimized pivot processing - only process a subset of pivots per update
if barstate.isconfirmed and bar_index % updateFrequency == 0 and array.size(pivotPrices) > 0
    pivotCount = array.size(pivotPrices)

    // Clear old bins periodically (less frequently)
    if bar_index % 200 == 0
        map.clear(binCenters)
        map.clear(binWeightedCounts)
        map.clear(binPriceSums)
        map.clear(binPriceCounts)
        map.clear(binPolarities)
        lastProcessedPivotIndex := 0
        lastProcessedPivotIndex

    // Process only recent pivots or a limited number per update
    startIdx = math.max(0, pivotCount - maxPivotsPerUpdate)
    endIdx = pivotCount - 1

    // Process limited number of recent pivots
    if startIdx <= endIdx
        for idx = startIdx to endIdx by 1
            price = array.get(pivotPrices, idx)
            strength = array.get(pivotStrengths, idx)
            volZPivot = array.get(pivotVolZs, idx)
            barIdx = array.get(pivotBarIndices, idx)
            polarity = array.get(pivotPolarities, idx)

            // Apply recency decay
            recencyDecay = math.exp(-0.001 * (bar_index - barIdx))
            weight = strength * math.max(0.1, 1 + volZPivot * 0.1) * recencyDecay

            // Skip very old or weak pivots to save computation
            if recencyDecay < 0.1 or weight < 0.5
                continue

            // Map to bin
            binIndex = int(price / binSize)

            if map.contains(binWeightedCounts, binIndex)
                currentCount = map.get(binWeightedCounts, binIndex)
                currentPriceSum = map.get(binPriceSums, binIndex)
                currentPriceCount = map.get(binPriceCounts, binIndex)

                map.put(binWeightedCounts, binIndex, currentCount + weight)
                map.put(binPriceSums, binIndex, currentPriceSum + price)
                map.put(binPriceCounts, binIndex, currentPriceCount + 1)
            else
                map.put(binCenters, binIndex, binIndex * binSize)
                map.put(binWeightedCounts, binIndex, weight)
                map.put(binPriceSums, binIndex, price)
                map.put(binPriceCounts, binIndex, 1)
                map.put(binPolarities, binIndex, polarity)

// Create zones from bins with sufficient support
f_create_zones_from_bins() =>
    zoneCreationCount = 0
    maxNewZones = 20 // Limit new zones per update

    if map.size(binWeightedCounts) > 0
        for [binIndex, weightedCount] in binWeightedCounts
            if zoneCreationCount >= maxNewZones
                break

            if map.contains(binPriceCounts, binIndex)
                priceCount = map.get(binPriceCounts, binIndex)
                polarity = map.get(binPolarities, binIndex)
                priceSum = map.get(binPriceSums, binIndex)

                if priceCount >= 2 and weightedCount >= 1.5
                    centerPrice = priceSum / priceCount
                    halfWidth = math.max(wBaseATR * atr64, binSize * 0.5)

                    // Add zone
                    array.push(zonePriceLows, centerPrice - halfWidth)
                    array.push(zonePriceHighs, centerPrice + halfWidth)
                    array.push(zoneScores, weightedCount)
                    array.push(zonePolarities, polarity)
                    array.push(zoneTouchCounts, 0)
                    array.push(zonePriorHitRates, 0.5)
                    array.push(zoneLastTouchBars, bar_index)
                    array.push(zoneCenters, centerPrice)
                    array.push(zoneHalfWidths, halfWidth)

                    zoneCreationCount := zoneCreationCount + 1
                    zoneCreationCount

// Simplified merge zones - limit merge operations
f_merge_zones() =>
    mergeZoneCount = array.size(zonePriceLows)
    if mergeZoneCount > 1
        merged = array.new<bool>()
        for k = 0 to mergeZoneCount - 1 by 1
            array.push(merged, false)

        mergeOperations = 0
        maxMergeOps = 10 // Limit merge operations

        i = 0
        while i < mergeZoneCount and mergeOperations < maxMergeOps
            if array.get(merged, i)
                i := i + 1
                continue

            a1 = array.get(zonePriceLows, i)
            a2 = array.get(zonePriceHighs, i)
            polarityA = array.get(zonePolarities, i)

            j = i + 1
            while j < mergeZoneCount and mergeOperations < maxMergeOps
                if array.get(merged, j)
                    j := j + 1
                    continue

                b1 = array.get(zonePriceLows, j)
                b2 = array.get(zonePriceHighs, j)
                polarityB = array.get(zonePolarities, j)

                // Only merge zones of same polarity
                if polarityA == polarityB and f_iou1d(a1, a2, b1, b2) > mergeIoU
                    // Merge j into i
                    newLow = math.min(a1, b1)
                    newHigh = math.max(a2, b2)
                    newCenter = (newLow + newHigh) * 0.5
                    newHalfWidth = (newHigh - newLow) * 0.5
                    newScore = array.get(zoneScores, i) + array.get(zoneScores, j)
                    newTouchCount = array.get(zoneTouchCounts, i) + array.get(zoneTouchCounts, j)

                    array.set(zonePriceLows, i, newLow)
                    array.set(zonePriceHighs, i, newHigh)
                    array.set(zoneScores, i, newScore)
                    array.set(zoneTouchCounts, i, newTouchCount)
                    array.set(zoneCenters, i, newCenter)
                    array.set(zoneHalfWidths, i, newHalfWidth)

                    array.set(merged, j, true)
                    mergeOperations := mergeOperations + 1
                    mergeOperations
                j := j + 1
                j
            i := i + 1
            i

        // Remove merged zones (backwards iteration)
        m = mergeZoneCount - 1
        while m >= 0
            if array.get(merged, m)
                array.remove(zonePriceLows, m)
                array.remove(zonePriceHighs, m)
                array.remove(zoneScores, m)
                array.remove(zonePolarities, m)
                array.remove(zoneTouchCounts, m)
                array.remove(zonePriorHitRates, m)
                array.remove(zoneLastTouchBars, m)
                array.remove(zoneCenters, m)
                array.remove(zoneHalfWidths, m)
            m := m - 1
            m

// Zone decay and cleanup
f_decay_zones() =>
    decayZoneCount = array.size(zonePriceLows)
    if decayZoneCount > 0
        // Apply score decay for untouched zones
        for n = 0 to decayZoneCount - 1 by 1
            lastTouchBar = array.get(zoneLastTouchBars, n)
            barsSinceTouch = bar_index - lastTouchBar

            if barsSinceTouch > H
                currentScore = array.get(zoneScores, n)
                decayFactor = math.exp(-0.01 * barsSinceTouch)
                array.set(zoneScores, n, currentScore * decayFactor)

        // Cap number of zones by removing lowest scoring ones
        while array.size(zonePriceLows) > maxZones
            lowestScore = array.get(zoneScores, 0)
            lowestIndex = 0

            for p = 1 to array.size(zoneScores) - 1 by 1
                score = array.get(zoneScores, p)
                if score < lowestScore
                    lowestScore := score
                    lowestIndex := p
                    lowestIndex

            // Remove lowest scoring zone
            array.remove(zonePriceLows, lowestIndex)
            array.remove(zonePriceHighs, lowestIndex)
            array.remove(zoneScores, lowestIndex)
            array.remove(zonePolarities, lowestIndex)
            array.remove(zoneTouchCounts, lowestIndex)
            array.remove(zonePriorHitRates, lowestIndex)
            array.remove(zoneLastTouchBars, lowestIndex)
            array.remove(zoneCenters, lowestIndex)
            array.remove(zoneHalfWidths, lowestIndex)

// Update zones with controlled frequency
if barstate.isconfirmed and bar_index % updateFrequency == 0
    // Clear old zones first
    array.clear(zonePriceLows)
    array.clear(zonePriceHighs)
    array.clear(zoneScores)
    array.clear(zonePolarities)
    array.clear(zoneTouchCounts)
    array.clear(zonePriorHitRates)
    array.clear(zoneLastTouchBars)
    array.clear(zoneCenters)
    array.clear(zoneHalfWidths)

    f_create_zones_from_bins()
    f_merge_zones()
    f_decay_zones()
    zonesUpdated := true
    zonesUpdated
else
    zonesUpdated := false
    zonesUpdated

// Touch detection and reaction probability calculation

    currentLow = low
    currentHigh = high
    currentClose = close

    for touchIdx = 0 to touchZoneCount - 1 by 1
        zoneLow = array.get(zonePriceLows, touchIdx)
        zoneHigh = array.get(zonePriceHighs, touchIdx)
        polarity = array.get(zonePolarities, touchIdx)
        center = array.get(zoneCenters, touchIdx)
        halfWidth = array.get(zoneHalfWidths, touchIdx)

        // Check if current bar touches zone
        if currentHigh >= zoneLow and currentLow <= zoneHigh
            // Update touch count and last touch bar
            touchCount = array.get(zoneTouchCounts, touchIdx) + 1
            array.set(zoneTouchCounts, touchIdx, touchCount)
            array.set(zoneLastTouchBars, touchIdx, bar_index)

            // Calculate reaction probability features
            distATR = math.abs(currentClose - center) / atr64
            widthATR = halfWidth / atr64
            vwapDiffATR = math.abs(currentClose - sessionVWAP) / atr64
            priorHitRate = array.get(zonePriorHitRates, touchIdx)

            // Wick penetration calculation
            if polarity > 0 // support zone
                wickPenATR := math.max(0, center - currentLow) / atr64
                wickPenATR
            else // resistance zone
                wickPenATR := math.max(0, currentHigh - center) / atr64
                wickPenATR

            // Trend flag
            trendFlag = adx14 > 25 ? -1.0 : 1.0

            // Logistic model calculation
            z = w0 + w1 * distATR + w2 * widthATR + w3 * volZ + w4 * vwapDiffATR + w5 * math.log(1 + touchCount) + w6 * priorHitRate + w7 * wickPenATR + w8 * trendFlag

            // Add mini-conv features if enabled
            if enableMiniConvs
                z := z + wgGrad * gradFilter + wgSmooth * smoothFilter + wgDoG * dogFilter
                z

            q = f_sigmoid(z)

            // Store for plotting and alerts
            currentQ := q
            currentTouchCount := touchCount
            currentPriorHitRate := priorHitRate
            currentTouchInfo := str.tostring(polarity > 0 ? 'Support' : 'Resistance') + ' @ ' + str.tostring(center, '#.##') + ' | Q=' + str.tostring(math.round(q * 100)) + '%' + ' | Touches=' + str.tostring(touchCount)

            // Update prior hit rate (simplified - in real implementation would track actual outcomes)
            if liveStrict
                // Only update with confirmed historical data
                // This is a placeholder - real implementation would track MFE/MAE outcomes
                newHitRate = f_ema_update(priorHitRate, q, 0.1)
                array.set(zonePriorHitRates, touchIdx, newHitRate)

            break // Only process first touched zone per bar

                    opacity = 100 - f_clamp(int(array.get(zoneScores, labelIdx) * 20), 0, 80)
                    zoneColor = isSupport ? color.new(color.green, opacity) : color.new(color.red, opacity)

                    labelText = showProbabilities ? 'Q: ' + str.tostring(math.round(currentQ * 100)) + '%' + '\nTouches: ' + str.tostring(touchCount) + '\nHit Rate: ' + str.tostring(math.round(priorHitRate * 100)) + '%' : 'Zone Touch'

                    labelY = isSupport ? zoneLow : zoneHigh
                    labelStyle = isSupport ? label.style_label_up : label.style_label_down

                    zoneLabel = label.new(bar_index, labelY, labelText, style = labelStyle, color = zoneColor, textcolor = color.white, size = size.small)

                    array.push(activeLabels, zoneLabel)
                break
