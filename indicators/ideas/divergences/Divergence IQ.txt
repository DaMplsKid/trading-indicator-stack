normalize() => 

    normSrc = src

    var srcArr = array.new<float>()

    srcArr.push(src)

    if srcArr.size() > window
        srcArr.shift()

    if normMethod == normalizationInput.rollingMinMax

        srcMax = srcArr.max(), srcMin = srcArr.min()

        normSrc := (src - srcMin) / (srcMax - srcMin)

    if normMethod == normalizationInput.rollingZscore

        mean = srcArr.avg(), stdDev = srcArr.stdev()

        normSrc := switch stdDev 

            0 => 0
            =>   (src - mean) / stdDev

    if normMethod == normalizationInput.robustScaling

        median = srcArr.median()

        devArr = array.new_float(srcArr.size(), na)

        for [i, data] in devArr 

            devArr.set(i, math.abs(srcArr.get(i) - median))

        mad = devArr.median()

        normSrc := switch mad != 0

            true => 0.6745 * (src - median) / mad  
            => 0


    normSrc

method add(map<string, float> id, string key, float value, bool isShort) =>

    finalValue = value

    if isShort

        finalValue *= -1

    id.put(key, nz(id.get(key)) + finalValue)

method tradeExits(map<string, float> tradeMap) => 


    if na(tradeMap.get(mapValues.direction))
        tradeMap.put(mapValues.direction, 0)

    if tradeMap.get(mapValues.direction) == 1

        isShort = false 

        getTarget = tradeMap.get(mapValues.target)
        getStop   = tradeMap.get(mapValues.stop)
        getEntry  = tradeMap.get(mapValues.entry)

        if open >= getTarget 

            tradeMap.addAll(true, getTarget, getEntry, getStop, isShort, exitData)

        else if open <= getStop 

            tradeMap.addAll(false, getTarget, getEntry, getStop, isShort, exitData)
            
        else if high >= getTarget 

            if low > getStop 

                tradeMap.addAll(true, getTarget, getEntry, getStop, isShort, exitData)

            else 

                if h.size() > 0

                    for i = 0 to h.size() - 1

                        ltfHigh = h.get(i), ltfLow = l.get(i), ltfClose = c.get(i), ltfOpen = o.get(i)

                        if ltfHigh >= getTarget 

                            if ltfLow > getStop 

                                tradeMap.addAll(true, getTarget, getEntry, getStop, isShort, exitData)

                            else 
                                
                                closer2low = math.abs(ltfOpen - ltfHigh) > math.abs(ltfOpen - ltfLow)

                                switch closer2low

                                    true => 
                            
                                         tradeMap.addAll(false, getTarget, getEntry, getStop, isShort, exitData)

                                    =>       

                                         tradeMap.addAll(true, getTarget, getEntry, getStop, isShort, exitData)
                            break 

                        else if ltfLow <= getStop 

                            tradeMap.addAll(false, getTarget, getEntry, getStop, isShort, exitData)

                            break 


                else 

                    closer2low = math.abs(open - high) > math.abs(open - low)

                    switch closer2low

                        true => 
                            
                             tradeMap.addAll(false, getTarget, getEntry, getStop, isShort, exitData)

                        =>       
                             tradeMap.addAll(true, getTarget, getEntry, getStop, isShort, exitData)

        else if low <= getStop

            tradeMap.addAll(false, getTarget, getEntry, getStop, isShort, exitData)
            

    else if tradeMap.get(mapValues.direction) == -1


        isShort = true

        getTarget = tradeMap.get(mapValues.target)
        getStop   = tradeMap.get(mapValues.stop)
        getEntry  = tradeMap.get(mapValues.entry)

        if open <= getTarget 

            tradeMap.addAll(true, getTarget, getEntry, getStop, isShort, exitData)

        else if open >= getStop 

            tradeMap.addAll(false, getTarget, getEntry, getStop, isShort, exitData)
            
        else if low <= getTarget 

            if high < getStop 

                tradeMap.addAll(true, getTarget, getEntry, getStop, isShort, exitData)

            else 

                if h.size() > 0

                    for i = 0 to h.size() - 1

                        ltfHigh = h.get(i), ltfLow = l.get(i), ltfClose = c.get(i), ltfOpen = o.get(i)

                        if ltfLow <= getTarget 

                            if ltfHigh < getStop 

                                tradeMap.addAll(true, getTarget, getEntry, getStop, isShort, exitData)

                            else 
                                
                                closer2low = math.abs(ltfOpen - ltfHigh) > math.abs(ltfOpen - ltfLow)

                                switch closer2low

                                    true => 
                            
                                         tradeMap.addAll(true, getTarget, getEntry, getStop, isShort, exitData)

                                    =>       

                                         tradeMap.addAll(false, getTarget, getEntry, getStop, isShort, exitData)
                            break 

                        else if ltfHigh >= getStop 

                            tradeMap.addAll(false, getTarget, getEntry, getStop, isShort, exitData)

                            break 


                else 

                    closer2low = math.abs(open - high) > math.abs(open - low)

                    switch closer2low

                        true => 
                            
                             tradeMap.addAll(true, getTarget, getEntry, getStop, isShort, exitData)

                        =>       
                             tradeMap.addAll(false, getTarget, getEntry, getStop, isShort, exitData)

        else if high >= getStop

            tradeMap.addAll(false, getTarget, getEntry, getStop, isShort, exitData)
            
    float(na)
