    int target_bar = bar_index - target_idx
    float res = 0.0
    
    if target_type == "Candle"
        res := close[target_idx] >= open[target_idx] ? 1.0 : -1.0
    else if target_type == "HTF Candle"
        res := htf_c[target_idx] >= htf_o[target_idx] ? 1.0 : -1.0
    else if target_type == "Pivot State"
        float p_val = 1.0 // Default baseline
        int sz = array.size(pivot_bars)
        if sz > 0
            bool found_future = false
            // Scan array to find the *first pivot that occurs after* the targeted historical bar
            for i = 0 to sz - 1
                if array.get(pivot_bars, i) > target_bar
                    // If the next upcoming pivot is a PH (-1), the current trend is an upward wave (1) leading to it.
                    // If it is an HH (-2), it is a strong upward wave (2).
                    // If it is a PL (1), the current trend is a downward wave (-1).
                    // If it is an LL (2), it is a strong downward wave (-2).
                    p_val := -array.get(pivot_types, i)
                    found_future := true
                    break
            
            // If the next pivot is not yet confirmed (still forming), extend the most recently confirmed state
            if not found_future
                p_val := array.get(pivot_types, sz - 1)
        res := p_val
    res

// Dense Layer: Performs matrix multiplication between inputs (o) and weights (w)
Dense(o, w, u) =>
    for j = 0 to array.size(u) - 1
        float sum = 0.0
        for i = 0 to array.size(o) - 1
            sum += array.get(o, i) * matrix.get(w, i, j)
        array.set(u, j, sum)


// ====================================================================================================
// [CONCEPT] Activation Functions
// Neural networks require non-linear transformations to learn complex, non-linear relationships
// within the data. Without activation functions, a deep neural network would mathematically 
// collapse into a simple linear regression model, regardless of how many layers it has.
//
// 1. ReLU (Rectified Linear Unit): f(x) = max(0, x)
//    - Replaces negative values with zero, keeping positive values unchanged.
//    - Extremely computationally efficient.
//    - Mitigates the "Vanishing Gradient" problem (where gradients become too small to update weights).
//    - In this code, a "Leaky ReLU" variant is used (f(x) = 0.01 * x for x < 0) to prevent 
//      "dying ReLU" (neurons permanently outputting zero and stopping learning).
//
// 2. Tanh (Hyperbolic Tangent): f(x) = (e^x - e^-x) / (e^x + e^-x)
//    - S-shaped curve squashing values into the range [-1, 1].
//    - Zero-centered, which generally makes the optimization process easier and faster than Sigmoid.
//    - Prone to vanishing gradients for very large positive or negative inputs.
//
// 3. Sigmoid: f(x) = 1 / (1 + e^-x)
//    - S-shaped curve squashing values into the range [0, 1].
//    - Historically popular, often used in the output layer for probability estimations.
//    - Suffers heavily from vanishing gradients and is not zero-centered.
//
// * Inverted Dropout: During training, some nodes are randomly "dropped" (set to 0) to prevent
//   overfitting. The remaining active nodes are scaled up by (1 / (1 - drop_rate)) to maintain 
//   the same expected mathematical sum, ensuring the scale of activations remains consistent during inference.
// ====================================================================================================
Activation(u, h, a, apply_dropout, drop_rate) =>
    float scale = apply_dropout ? 1.0 / (1.0 - drop_rate) : 1.0
    for i = 0 to array.size(u)-1
        val = array.get(u, i)
        float res = 0.0
        if a == "ReLU"
            res := val > 0 ? val : val * 0.01 // Leaky ReLU
        else if a == "Tanh"
            res := (math.exp(val)-math.exp(-val))/(math.exp(val)+math.exp(-val))
        else // Sigmoid
            res := 1/(1+math.exp(-val))
            
        // Inverted Dropout Implementation
        if apply_dropout and math.random(0.0, 1.0) < drop_rate
            res := 0.0 // Node is dropped
        else
            res := res * scale // Scale up active nodes to maintain expected sum
            
        array.set(h, i, res)
    array.set(h, array.size(h)-1, 1.0) // Set Bias node to 1.0

// Layer Normalization: Standardizes node values within a layer to stabilize training
LayerNorm(u, apply_ln) =>
    if apply_ln
        float mean = array.avg(u)
        float std = array.stdev(u)
        if std == 0.0 or na(std)
            std := 1.0
        for i = 0 to array.size(u) - 1
            array.set(u, i, (array.get(u, i) - mean) / std)

// Derivative of the Activation Function (used for backpropagation gradients)
Deriv(h, i, a) =>
    val = array.get(h, i)
    float deriv = 0.0
    if val == 0.0 // Cut off gradient flow for dropped-out nodes
        deriv := 0.0
    else
        if a == "ReLU"
            deriv := val > 0 ? 1.0 : 0.01
        else if a == "Tanh"
            deriv := 1.0 - math.pow(val, 2)
        else // Sigmoid
            deriv := (1.0 - val) * val
    deriv

// [8] Optimizer Functions

// ====================================================================================================
// [CONCEPT] Gradient Descent & Optimizers
// The goal of training a neural network is to find the optimal set of weights that minimizes the 
// loss function (the error between predictions and actual targets). Backpropagation calculates the 
// "gradient" (derivative), which indicates the direction to adjust the weights to reduce the error.
// Optimizers determine *how* the weights are updated based on these gradients.
//
// 1. SGD (Stochastic Gradient Descent)
//    - Update Rule: W_new = W_old - (Learning_Rate * Gradient)
//    - The most basic approach. It updates weights directly proportional to the current gradient.
//    - Prone to oscillation in ravines (areas where the loss surface curves much more steeply in 
//      one dimension than another) and can easily get stuck in local minima.
//
// 2. Momentum
//    - Conceptually similar to a heavy ball rolling down a hill; it accumulates "momentum" in directions 
//      where gradients consistently point, while dampening oscillations in other directions.
//    - Computes a moving average of past gradients (1st moment, 'm').
//    - Helps accelerate SGD in the relevant direction.
//
// 3. RMSprop (Root Mean Square Propagation)
//    - Addresses the issue of a global learning rate by adapting the learning rate for each weight individually.
//    - It keeps a moving average of the *squared* gradients (2nd moment, 'v_m').
//    - Weights with large recent gradients get a smaller effective learning rate, and vice versa. 
//      This prevents exploding gradients and handles non-stationary objectives well.
//
// 4. Adam (Adaptive Moment Estimation)
//    - Combines the best properties of both Momentum (1st moment 'm') and RMSprop (2nd moment 'v_m').
//    - 'm' tracks the direction of recent gradients, while 'v_m' tracks the magnitude.
//    - Adam dynamically adjusts the learning rate for each individual weight.
//    - Includes "Bias Correction" (m_hat, v_hat) to counteract the fact that m and v are initialized to zero,
//      which otherwise causes an initial bias toward zero, especially during early training steps.
//    - Generally considered the most robust and standard choice for deep learning applications.
// ====================================================================================================
Update_Weights(w, dw, m_w, v_w, opt_type, step) =>
    float beta1 = 0.9
    float beta2 = 0.999
    float epsilon = 1e-8
    
    for i = 0 to matrix.rows(w) - 1
        for j = 0 to matrix.columns(w) - 1
            float g = matrix.get(dw, i, j)
            float current_w = matrix.get(w, i, j)
            
            // Apply L1 & L2 Regularization (Skip the Bias row, which is the last row)
            if i < matrix.rows(w) - 1
                if l1_reg > 0.0
                    g += l1_reg * math.sign(current_w) // L1 Penalty
                if l2_reg > 0.0
                    g += l2_reg * current_w            // L2 Penalty

            float new_w = current_w
            
            if opt_type == "SGD"
                new_w := current_w - lr * g
            else if opt_type == "Momentum"
                float m = matrix.get(m_w, i, j)
                float new_m = beta1 * m + lr * g
                matrix.set(m_w, i, j, new_m)
                new_w := current_w - new_m
            else if opt_type == "RMSprop"
                float v_m = matrix.get(v_w, i, j)
                float new_v = beta1 * v_m + (1.0 - beta1) * math.pow(g, 2)
                matrix.set(v_w, i, j, new_v)
                new_w := current_w - lr * g / (math.sqrt(new_v) + epsilon)
            else if opt_type == "Adam"
                float m = matrix.get(m_w, i, j)
                float v_m = matrix.get(v_w, i, j)
                float new_m = beta1 * m + (1.0 - beta1) * g
                float new_v = beta2 * v_m + (1.0 - beta2) * math.pow(g, 2)
                matrix.set(m_w, i, j, new_m)
                matrix.set(v_w, i, j, new_v)
                
                // Bias correction for Adam (compensates for initialization at zero)
                float m_hat = new_m / (1.0 - math.pow(beta1, step))
                float v_hat = new_v / (1.0 - math.pow(beta2, step))
                new_w := current_w - lr * m_hat / (math.sqrt(v_hat) + epsilon)
                
            matrix.set(w, i, j, new_w)

// -----------------------------------------------------------------------------
// [9] Engine: Online Training Loop (Executes on every bar)
// -----------------------------------------------------------------------------
for e = 0 to epoch
    // Randomly sample a historical bar within the window (excluding bars that don't have a realized target yet)
    t = int(math.random(predict_period + 1, sampling_window))
    float[] feats = get_features(t)
    
    // Load features into the input array
    for i = 0 to layer_1 - 1
        array.set(x, i, nz(array.get(feats, i), 0.0))
    array.set(x, layer_1, 1.0) // Set Bias input

    // Forward Propagation (Training phase: Dropout is ON)
    for i = 0 to num_hidden_layers - 1
        Layer L = array.get(layers, i)
        array<float> prev_input = (i == 0) ? x : array.get(layers, i-1).h
        Dense(prev_input, L.w, L.u)
        LayerNorm(L.u, use_layer_norm)
        Activation(L.u, L.h, activation, true, dropout_rate)
    Layer last_L = array.get(layers, num_hidden_layers-1)
    Dense(last_L.h, v, p)

    // Calculate prediction error
    float delta_out = array.get(p, 0) - get_target(t)
    
    // Backpropagation: Output Layer Delta
    for i = 0 to array.size(last_L.delta)-1
        array.set(last_L.delta, i, delta_out * matrix.get(v, i, 0) * Deriv(last_L.h, i, activation))
    
    // Backpropagation: Hidden Layers Deltas
    if num_hidden_layers > 1
        for i = num_hidden_layers-2 to 0
            Layer curr = array.get(layers, i)
            Layer next = array.get(layers, i+1)
            for j = 0 to array.size(curr.delta)-1
                float err = 0.0
                for k = 0 to array.size(next.delta)-1
                    err += array.get(next.delta, k) * matrix.get(next.w, j, k)
                array.set(curr.delta, j, err * Deriv(curr.h, j, activation))
    
    // Update Weights Using Optimizer
    global_step += 1 // Increment step count for Adam bias correction
    
    // Compute gradients for Output Layer
    for i = 0 to matrix.rows(v)-1
        matrix.set(dv, i, 0, delta_out * array.get(last_L.h, i))
    Update_Weights(v, dv, m_v, v_v, optimizer, global_step)

    // Compute gradients for Hidden Layers
    for k = 0 to num_hidden_layers-1
        Layer L = array.get(layers, k)
        array<float> prev_update = (k == 0) ? x : array.get(layers, k-1).h
        for i = 0 to matrix.rows(L.w)-1
            for j = 0 to matrix.columns(L.w)-1
                matrix.set(L.dw, i, j, array.get(L.delta, j) * array.get(prev_update, i))
        
        Update_Weights(L.w, L.dw, L.m_w, L.v_w, optimizer, global_step)

// -----------------------------------------------------------------------------
// [10] Current Inference (Forward Feed for Prediction)
// -----------------------------------------------------------------------------
float[] curr_feats = get_features(0)
for i = 0 to layer_1 - 1
    array.set(x, i, nz(array.get(curr_feats, i), 0.0))
array.set(x, layer_1, 1.0)

// Forward pass for the current bar (Inference phase: Dropout is OFF)
for i = 0 to num_hidden_layers - 1
    Layer L = array.get(layers, i)
    array<float> prev_inf = (i == 0) ? x : array.get(layers, i-1).h
    Dense(prev_inf, L.w, L.u)
    LayerNorm(L.u, use_layer_norm)
    Activation(L.u, L.h, activation, false, 0.0) 
Layer final_L = array.get(layers, num_hidden_layers-1)
Dense(final_L.h, v, p)

// -----------------------------------------------------------------------------
// [11] Statistical Post-Processing & Kalman Filter Application
// -----------------------------------------------------------------------------
float pred_raw = array.get(p, 0)
// Smooth the standard deviation of raw predictions to compute a stable Z-score multiplier
float stdev_p = ta.median(ta.stdev(pred_raw, 500),500)
float scale_factor = stdev_p != 0 ? 1.0 / stdev_p : 0.0

// Convert prediction into a standardized Z-Score and clamp it between -3.0 and 3.0
float NN_value = pred_raw * scale_factor
NN_value := NN_value > 3.0 ? 3.0 : NN_value < -3.0 ? -3.0 : NN_value

// Apply 1D Kalman Filter to smooth the final signal
if apply_kalman
    float p_err = kf_err + kf_q
    float k_gain = p_err / (p_err + kf_r)
    kf_est := kf_est + k_gain * (NN_value - kf_est)
    kf_err := (1.0 - k_gain) * p_err
    NN_value := kf_est
else
    kf_est := NN_value

// -----------------------------------------------------------------------------
// [12] MC Dropout (Uncertainty Estimation) & Confidence Box Plotting
// -----------------------------------------------------------------------------
float ci_margin = 0.0

// Calculate uncertainty only on the latest bar to save computation resources
if barstate.islast
    array<float> mc_preds = array.new<float>()
    // Run multiple forward passes with Dropout enabled
    for m = 0 to mc_samples - 1
        for i = 0 to num_hidden_layers - 1
            Layer L_mc = array.get(layers, i)
            array<float> prev_inf_mc = (i == 0) ? x : array.get(layers, i-1).h
            Dense(prev_inf_mc, L_mc.w, L_mc.u)
            LayerNorm(L_mc.u, use_layer_norm)
            Activation(L_mc.u, L_mc.h, activation, true, dropout_rate) // Dropout ON
        
        Layer final_L_mc = array.get(layers, num_hidden_layers-1)
        array<float> p_mc = array.new<float>(1, 0.0)
        Dense(final_L_mc.h, v, p_mc)
        array.push(mc_preds, array.get(p_mc, 0))
    
    // Calculate the margin based on the standard deviation of MC predictions
    float mc_std = array.stdev(mc_preds)
    ci_margin := (mc_std * ci_z_score) * scale_factor
    
    // Restore the network state to deterministic mode (Dropout OFF) for dashboard UI display
    for i = 0 to num_hidden_layers - 1
        Layer L_rest = array.get(layers, i)
        array<float> prev_inf_rest = (i == 0) ? x : array.get(layers, i-1).h
        Dense(prev_inf_rest, L_rest.w, L_rest.u)
        LayerNorm(L_rest.u, use_layer_norm)
        Activation(L_rest.u, L_rest.h, activation, false, 0.0)

    // Draw the Confidence Interval Box projected into the future (avoids overlapping the oscillator)
    color box_bg = NN_value > 0 ? color.new(color_up, 85) : color.new(color_dn, 85)

    else

// Helper function to dynamically calculate color opacity based on standard deviations
get_dynamic_color(val, stdev_val, min_alpha, max_alpha) =>
    // Uses 2x the standard deviation (2 sigma) as the upper bound for color scaling
    float max_val = stdev_val * 2.0
    float ratio = max_val == 0 ? 0 : math.min(1.0, math.max(0.0, math.abs(val) / max_val))
    float alpha = max_alpha - (ratio * (max_alpha - min_alpha))
    val > 0 ? color.new(color_up, int(alpha)) : color.new(color_dn, int(alpha))

// Helper function to colorize hidden layer nodes in the dashboard
get_hidden_color(val, act_type) =>
    float norm_val = val
    float max_v = 1.0
    if act_type == "Sigmoid"
        norm_val := val - 0.5 // Center sigmoid
        max_v := 0.5
    else if act_type == "ReLU"
        norm_val := val - 0.5
        max_v := 1.0
    
    float ratio = math.min(1.0, math.max(0.0, math.abs(norm_val) / max_v))
    float alpha = 90.0 - (ratio * 70.0) // Ranges from 20 (Dark/Strong) to 90 (Light/Weak)
    norm_val > 0 ? color.new(color_up, int(alpha)) : color.new(color_dn, int(alpha))
