// ▌ CORE RANGE FILTER ─────────────────────────────────────────────────────
f_range_filter(src, sens, atr_len, atr_mult) =>
    atr   = ta.atr(atr_len)
    rng   = atr * atr_mult * (sens / 8.0)
    var float filt  = src
    var int   trend = 0
    filt  := src > filt + rng ? src - rng : src < filt - rng ? src + rng : filt
    prev   = trend[1]
    trend := filt > filt[1] ? 1 : filt < filt[1] ? -1 : trend[1]
    sig    = trend != prev ? trend : 0
    [filt, trend, sig]

[rf_filt, rf_trend, rf_sig] = f_range_filter(close, i_sensitivity, i_atr_len, i_atr_mult)

// ▌ ATR & VOLATILITY ──────────────────────────────────────────────────────
atr_val  = ta.atr(i_atr_len)
atr_norm = atr_val / close * 100

atr_rank = 0.0
for k = 0 to 59
    if atr_val >= atr_val[k]
        atr_rank += 1
atr_rank := math.round(atr_rank / 60 * 100)

vol_label = atr_rank > 75 ? "HIGH" : atr_rank > 45 ? "MED" : "LOW"
vol_col   = atr_rank > 75 ? c_bear : atr_rank > 45 ? DASH_GOLD : c_bull
score_vol = (atr_rank < 80 ? 1 : 0) + (atr_norm < ta.percentile_nearest_rank(atr_norm, 60, 65) ? 1 : 0)

// ▌ MOMENTUM ──────────────────────────────────────────────────────────────
roc5  = (close - close[5])  / close[5]  * 100
roc10 = (close - close[10]) / close[10] * 100
roc20 = (close - close[20]) / close[20] * 100

mom_bull    = roc5 > 0 and roc10 > 0 and roc20 > 0
mom_bear    = roc5 < 0 and roc10 < 0 and roc20 < 0
mom_aligned = (rf_trend == 1 and mom_bull) or (rf_trend == -1 and mom_bear)
mom_partial = (rf_trend == 1 and roc5 > 0) or (rf_trend == -1 and roc5 < 0)

// ▌ ANTI-CHOP: ADX ────────────────────────────────────────────────────────
f_adx(len) =>
    up   = high - high[1]
    down = low[1] - low
    plus_dm  = up > down and up > 0 ? up : 0.0
    minus_dm = down > up and down > 0 ? down : 0.0
    true_range = ta.atr(1)
    plus_di  = 100 * ta.rma(plus_dm,  len) / ta.rma(true_range, len)
    minus_di = 100 * ta.rma(minus_dm, len) / ta.rma(true_range, len)
    dx       = math.abs(plus_di - minus_di) / (plus_di + minus_di) * 100
    adx_val  = ta.rma(dx, len)
    [adx_val, plus_di, minus_di]

[adx_value, di_plus, di_minus] = f_adx(i_adx_len)
adx_trending = adx_value >= i_adx_thresh

// ▌ ANTI-CHOP: Choppiness Index ───────────────────────────────────────────
chop_atr_sum = math.sum(ta.atr(1), i_chop_len)
chop_hi      = ta.highest(high, i_chop_len)
chop_lo      = ta.lowest(low, i_chop_len)
chop_range   = chop_hi - chop_lo
chop_index   = chop_range > 0 ? 100 * math.log10(chop_atr_sum / chop_range) / math.log10(i_chop_len) : 50.0
chop_clear   = chop_index <= i_chop_thresh

// ▌ ANTI-CHOP: Signal Cooldown ────────────────────────────────────────────
var int bars_since_signal = 999
bars_since_signal := bars_since_signal + 1
cooldown_clear = not i_use_cooldown or bars_since_signal >= i_cooldown_len

// ▌ ANTI-CHOP: Combined Gate ──────────────────────────────────────────────
chop_gate    = (not i_use_adx or adx_trending) and (not i_use_chop or chop_clear) and cooldown_clear
chop_penalty = (i_use_adx and not adx_trending ? -1 : 0) + (i_use_chop and not chop_clear ? -1 : 0)

// ▌ EMA OVERLAY + ALIGNMENT ───────────────────────────────────────────────

ema_price_above = close > ema_fast
ema_slope_up    = ema_fast > ema_fast[5]
ema_all_bull    = ema_cross_bull and ema_price_above and ema_slope_up

ema_cross_bear  = ema_fast < ema_slow
ema_price_below = close < ema_fast
ema_slope_down  = ema_fast < ema_fast[5]
ema_all_bear    = ema_cross_bear and ema_price_below and ema_slope_down

// ▌ PIVOT LEVELS (S1–S4 / P / R1–R4) ──────────────────────────────────────
// Pulls prior-period H/L/C from the selected timeframe. Levels are stationary
// until the period rolls over. Works on any instrument and any chart timeframe.

[pvt_h, pvt_l, pvt_c] = request.security(syminfo.tickerid,i_pvt_tf,[high[1], low[1], close[1]],lookahead=barmerge.lookahead_on)

pvt_pp    = (pvt_h + pvt_l + pvt_c) / 3.0
pvt_range = pvt_h - pvt_l

// ── Classic ──
f_classic_r(n) =>
    n == 1 ? 2.0 * pvt_pp - pvt_l :n == 2 ? pvt_pp + pvt_range :n == 3 ? pvt_h + 2.0 * (pvt_pp - pvt_l) :pvt_pp + pvt_range * 3.0

f_classic_s(n) =>
    n == 1 ? 2.0 * pvt_pp - pvt_h :n == 2 ? pvt_pp - pvt_range :n == 3 ? pvt_l - 2.0 * (pvt_h - pvt_pp) :pvt_pp - pvt_range * 3.0

// ── Fibonacci ──
f_fib_r(n) =>
    n == 1 ? pvt_pp + pvt_range * 0.382 :n == 2 ? pvt_pp + pvt_range * 0.618 :n == 3 ? pvt_pp + pvt_range * 1.000 : pvt_pp + pvt_range * 1.618

f_fib_s(n) =>
    n == 1 ? pvt_pp - pvt_range * 0.382 : n == 2 ? pvt_pp - pvt_range * 0.618 :n == 3 ? pvt_pp - pvt_range * 1.000 :pvt_pp - pvt_range * 1.618
// ── Camarilla ──
f_cam_r(n) =>
    n == 1 ? pvt_c + pvt_range * 1.1 / 12.0 :n == 2 ? pvt_c + pvt_range * 1.1 / 6.0  :n == 3 ? pvt_c + pvt_range * 1.1 / 4.0  :pvt_c + pvt_range * 1.1 / 2.0

f_cam_s(n) =>
    n == 1 ? pvt_c - pvt_range * 1.1 / 12.0 :n == 2 ? pvt_c - pvt_range * 1.1 / 6.0  :n == 3 ? pvt_c - pvt_range * 1.1 / 4.0  :pvt_c - pvt_range * 1.1 / 2.0

f_r(n) =>
    i_pvt_type == "Classic" ? f_classic_r(n) :i_pvt_type == "Fibonacci" ? f_fib_r(n) :f_cam_r(n)

f_s(n) =>
    i_pvt_type == "Classic" ? f_classic_s(n) :i_pvt_type == "Fibonacci" ? f_fib_s(n) :f_cam_s(n)
