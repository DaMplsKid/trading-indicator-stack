//@version=5
// Voss Predictive Filter (VPF) with JMA-powered Z-score pivots
indicator("Voss Predictive Filter (VPF) + JMA Z-score Pivots", "VPF_JMAZPivots", false)

// === Inputs ===
src = input.source(close, "Source")
len = input.int(20, "Length")
pre = input.int(3, "Predict")
band = input.float(0.25, "Bandwidth")

// Custom Pivot Inputs for Z-Score
pivot_high_lookback = input.int(10, "Pivot High Lookback", minval=1)
pivot_low_lookback  = input.int(10, "Pivot Low Lookback", minval=1)
zscore_lookback     = input.int(20, "Z-Score Lookback", minval=5)
zscore_threshold    = input.float(1.5, "Z-Score Strength Threshold", minval=0.5, step=0.1)

// JMA params for Z-score mean
jma_zlen   = input.int(20, "Z-Score JMA Length", minval=5)
jma_zphase = input.int(50, "Z-Score JMA Phase", minval=-100, maxval=100)
jma_zpower = input.int(2, "Z-Score JMA Power", minval=1, maxval=4)

// Pivot shapes/colors
pivot_shape_high = input.string(title="Pivot High Shape", defval="Triangle Down", options=["Triangle Down", "Cross", "Circle", "Arrow Down"])
pivot_shape_low  = input.string(title="Pivot Low Shape", defval="Triangle Up", options=["Triangle Up", "Cross", "Circle", "Arrow Up"])
pivot_color_high = input.color(color.purple, "Pivot High Color")
pivot_color_low  = input.color(color.navy, "Pivot Low Color")

// === JMA Function ===
f_jma(src, length, phase, power) =>
    phaseRatio = phase < -100 ? 0.5 : phase > 100 ? 2.5 : phase / 100 + 1.5
    beta = 0.45 * (length - 1) / (0.45 * (length - 1) + 2)
    alpha = math.pow(beta, power)
    var float jma_val = na
    var float e0 = na
    var float e1 = na
    var float e2 = na
    e0 := (1 - alpha) * src + alpha * nz(e0[1], src)
    e1 := (src - e0) * (1 - beta) + beta * nz(e1[1], 0)
    e2 := (e0 + phaseRatio * e1 - nz(jma_val[1], src)) * math.pow(1 - alpha, 2) + math.pow(alpha, 2) * nz(e2[1], 0)
    jma_val := e2 + nz(jma_val[1], src)
    jma_val

// === Voss Predictive Filter Function (V5 Conversion) ===
f_vpf(_len, _pre, _band, _src) =>
    var float _filt = 0.0, var float _voss = 0.0
    _pi  = 2 * math.asin(1)
    _ord = 3 * _pre
    _f1  = math.cos(2 * _pi / _len)
    _g1  = math.cos(_band * 2 * _pi / _len)
    _s1  = 1 / _g1 - math.sqrt(1 / (_g1 * _g1) - 1)
    _s2  = 1 + _s1
    _s3  = 1 - _s1
    _x1  = _src - _src[2]
    _x2  = (3 + _ord) / 2
    if bar_index > _ord
        _sumC = 0.0
        for _i = 0 to (_ord - 1)
            _sumC := _sumC + ((_i + 1) / _ord) * _voss[_ord - _i]
        _filt := 0.5 * _s3 * _x1 + _f1 * _s2 * _filt[1] - _s1 * _filt[2]
        _voss := _x2 * _filt - _sumC
    [_voss, _filt]

[voss, filt] = f_vpf(len, pre, band, src)

// === Z-Score Pivot Point Logic applied to Voss (uses JMA mean) ===
pivot_high = ta.pivothigh(voss, pivot_high_lookback, pivot_high_lookback)
pivot_low  = ta.pivotlow(voss, pivot_low_lookback, pivot_low_lookback)

// Calculate pivot strength (absolute move leading into the pivot)
ph_strength = not na(pivot_high) ? math.abs(voss[pivot_high_lookback] - voss[pivot_high_lookback + 1]) : na
pl_strength = not na(pivot_low)  ? math.abs(voss[pivot_low_lookback]  - voss[pivot_low_lookback + 1])  : na

// --- JMA Z-score function ---
zscore_jma(src, len, phase, power) =>
    m = f_jma(src, len, phase, power)
    d = ta.stdev(src, len)
    d == 0 ? na : (src - m) / d

// Calculate JMA Z-Score for pivot strengths
pivot_high_zscore = not na(ph_strength) ? zscore_jma(ph_strength, zscore_lookback, jma_zphase, jma_zpower) : na
pivot_low_zscore  = not na(pl_strength) ? zscore_jma(pl_strength,  zscore_lookback, jma_zphase, jma_zpower) : na

// Thresholding
show_pivot_high = not na(pivot_high_zscore) and math.abs(pivot_high_zscore) >= zscore_threshold
show_pivot_low  = not na(pivot_low_zscore)  and math.abs(pivot_low_zscore)  >= zscore_threshold

// === Helper function to get shape style from input string ===
get_shape_style(shape_string, is_high) =>
    switch shape_string
        "Triangle Down" => shape.triangledown
        "Triangle Up"   => shape.triangleup
        "Cross"         => shape.cross
        "Circle"        => shape.circle
        "Arrow Down"    => shape.arrowdown
        "Arrow Up"      => shape.arrowup
        => is_high ? shape.triangledown : shape.triangleup

// === Plots ===
plot(voss, "Voss", color.blue, 2, plot.style_line)
plot(filt, "Filter", color.red, 2, plot.style_line)
hline(0.0, "Zero", color.gray, hline.style_dashed, 1)

// Plot pivot shapes that pass z-score threshold, with user-selected color and shape
plotshape(series=show_pivot_high, title="Pivot High", location=location.top, style=get_shape_style(pivot_shape_high, true), color=pivot_color_high, size=size.small, offset=-pivot_high_lookback)
plotshape(series=show_pivot_low,  title="Pivot Low",  location=location.bottom, style=get_shape_style(pivot_shape_low, false), color=pivot_color_low, size=size.small, offset=-pivot_low_lookback)

// Plot the actual z-score value at each strong pivot as a label
if show_pivot_high
    label.new(bar_index - pivot_high_lookback, voss[pivot_high_lookback], "Z: " + str.tostring(pivot_high_zscore, format.mintick), style=label.style_label_down, color=pivot_color_high, textcolor=color.white, size=size.tiny)
if show_pivot_low
    label.new(bar_index - pivot_low_lookback, voss[pivot_low_lookback], "Z: " + str.tostring(pivot_low_zscore, format.mintick), style=label.style_label_up, color=pivot_color_low, textcolor=color.white, size=size.tiny)
