//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
// REGIME CALCULATION FUNCTION
//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
calcRegime(series float srcData, series float volumeData, int pMadLen, int pMadSlopeLen, int pPressureLen, float pPressureThresh, int pBwRocLen, int pEffLen, int pSmoothLen) =>
    
    // 1️⃣ MAD-BASED VOLATILITY
    medianPrice = ta.percentile_linear_interpolation(srcData, pMadLen, 50)
    madRaw      = ta.percentile_linear_interpolation(math.abs(srcData - medianPrice), pMadLen, 50)
    madVol      = 1.4826 * madRaw
    
    madSlope = ta.linreg(madVol, pMadSlopeLen, 0)
    
    // 2️⃣ PRESSURE MODEL
    roc        = ta.roc(srcData, 1)
    volNorm    = volumeData / ta.sma(volumeData, pPressureLen)
    pressure   = roc * volNorm
    
    pressureSmooth = ta.ema(pressure, pSmoothLen)
    pressureNorm = pressureSmooth / ta.stdev(pressureSmooth, pPressureLen)
    
    // 3️⃣ ADAPTIVE PRESSURE BANDS
    pressureFactor = math.min(math.max(math.abs(pressureNorm), 0.5), 2.0)
    
    upperBand = medianPrice + madVol * pressureFactor
    lowerBand = medianPrice - madVol * pressureFactor
    
    bandWidth = upperBand - lowerBand
    bwRoc = ta.roc(bandWidth, pBwRocLen)
    
    // 4️⃣ DIRECTIONAL EFFICIENCY (KER)
    netMove   = math.abs(srcData - srcData[pEffLen])
    grossMove = ta.sma(math.abs(srcData - srcData[1]), pEffLen) * pEffLen
    
    effRatio  = grossMove != 0 ? netMove / grossMove : 0
    
    // 5️⃣ REGIME LOGIC
    volExpanding   = madSlope > 0
    volContracting = madSlope <= 0
    
    intentStrong   = bwRoc > 0 and pressureNorm > pPressureThresh
    intentWeak     = bwRoc <= 0
    
    efficientMove  = effRatio > 0.35
    inefficient    = effRatio <= 0.35
    
    // 6️⃣ REGIME CLASSIFICATION
    trendIgnition = volExpanding and intentStrong and efficientMove
    fakeBreakout = volContracting and intentStrong and inefficient
    meanReversion = volContracting and intentWeak and inefficient
    volatilityChurn = volExpanding and intentStrong and inefficient
    
    // Return all values as tuple
    [madVol, madSlope, bwRoc, effRatio, trendIgnition, fakeBreakout, meanReversion, volatilityChurn, pressureNorm]

//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
// CURRENT TIMEFRAME CALCULATION
//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

     calcRegime(src, volume, madLen, madSlopeLen, pressureLen, pressureThresh, bwRocLen, effLen, smoothLen)

//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
