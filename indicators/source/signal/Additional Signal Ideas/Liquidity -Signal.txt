// HELPERS
//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

f_sum_abs_change(_src, _len) =>
    float s = 0.0
    for i = 0 to _len - 1
        s += math.abs(_src[i] - _src[i + 1])
    s

f_mad(_src, _len) =>
    ta.sma(math.abs(_src - ta.sma(_src, _len)), _len)

f_zscore(_src, _len) =>
    m = ta.sma(_src, _len)
    s = ta.stdev(_src, _len)
    s == 0.0 ? 0.0 : (_src - m) / s

jma_func(_src, _length, _phase, _power) =>
    phaseRatio = _phase < -100 ? 0.5 : _phase > 100 ? 2.5 : _phase / 100 + 1.5
    beta  = 0.45 * (_length - 1) / (0.45 * (_length - 1) + 2)
    alpha = math.pow(beta, _power)
    var float e0 = na
    var float e1 = na
    var float e2 = na
    var float jv = na
    e0 := (1 - alpha) * _src + alpha * nz(e0[1], _src)
    e1 := (_src - e0) * (1 - beta) + beta * nz(e1[1], 0)
    e2 := (e0 + phaseRatio * e1 - nz(jv[1], _src)) * math.pow(1 - alpha, 2) + math.pow(alpha, 2) * nz(e2[1], 0)
    jv  := e2 + nz(jv[1], _src)
    jv

//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
// LAYER 0 — JMA HEALTH (Background Color Driver)
//
//  jmaSlope    = first derivative  (is the JMA rising or falling?)
//  jmaAccel    = second derivative (is the slope gaining or losing momentum?)
//
//  State matrix:
//    Slope+ / Accel+  →  GREEN   (trend strengthening)
//    Slope+ / Accel-  →  YELLOW  (trend weakening, still up)
//    Slope- / Accel-  →  RED     (trend falling and accelerating down)
//    Slope- / Accel+  →  BLUE    (downtrend slowing, potential base)
//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

jma         = jma_func(close, jmaLen, jmaPhase, jmaPower)
jmaSlope    = jma - jma[slopeLen]

jmaAccel    = ta.ema(jmaAccelRaw, accelLen)   // Smoothed to remove single-bar noise

slopePos    = jmaSlope > 0
slopeNeg    = jmaSlope < 0
accelPos    = jmaAccel > 0
accelNeg    = jmaAccel < 0

// Four states
jmaGreen  = slopePos and accelPos    // Strengthening uptrend
jmaYellow = slopePos and accelNeg    // Weakening uptrend
jmaRed    = slopeNeg and accelNeg    // Accelerating downtrend
jmaBlue   = slopeNeg and accelPos    // Slowing downtrend

// LAYER 0b — VER + VWER (computed early — needed for divergence gate)
//
//  Must run before divergence section so efficiency state is available.
//  VER  = volatility-weighted efficiency ratio (penalises choppy path)
//  VWER = volume-weighted efficiency ratio     (penalises low participation)
//  Spread = VER - VWER  →  positive = chop dominant, negative = volume deficit
//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

ver_net     = math.abs(close - close[verEffLen])
ver_path    = f_sum_abs_change(close, verEffLen)
ver_path    := ver_path == 0 ? 1.0 : ver_path
ver_pathVol = f_mad(close, verPathLen)
ver_pathVol := ver_pathVol == 0 ? 1.0 : ver_pathVol
ver_avgVol  = ta.sma(volume, verVolLen)
ver_normVol = ver_avgVol != 0 ? volume / ver_avgVol : 1.0

ver  = ta.ema(ver_net / (ver_path * ver_pathVol), verSmooth)
vwer = ta.ema(ver_net / (ver_path * ver_normVol), verSmooth)

// Spread analytics (used in score modifier — Improvement 4)
verSpread      = ver - vwer                        // + = chop dominant, − = volume deficit
verSpreadSlope = verSpread - verSpread[3]          // 3-bar slope for stability vs noise
verSpreadChopDom  = verSpread > 0                  // Chop penalty > volume penalty
verSpreadVolDom   = verSpread < 0                  // Volume penalty > chop penalty

//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
// LAYER 1 — JMA PRICE DIVERGENCE (Arrows)
//
//  Divergence = price makes a new swing extreme but JMA does NOT confirm it.
//  This fires *before* structure breaks — the JMA is already disagreeing
//  with what price is doing.
//
//  Bearish divergence: price makes higher high, JMA makes lower high
//    → JMA is not confirming the new high → downward arrow
//
//  Bullish divergence: price makes lower low, JMA makes higher low
//    → JMA is not confirming the new low → upward arrow
//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

// Price pivots (used for divergence comparison only — not for structure)
pricePH = ta.pivothigh(high, divLookback, divLookback)
pricePL = ta.pivotlow(low,  divLookback, divLookback)

// JMA value at the same pivot bars
jmaPH = ta.pivothigh(jma, divLookback, divLookback)
jmaPL = ta.pivotlow(jma,  divLookback, divLookback)

// Track last two pivot values for both price and JMA
var float lastPriceHigh = na
var float prevPriceHigh = na
var float lastJMAHigh   = na
var float prevJMAHigh   = na

var float lastPriceLow  = na
var float prevPriceLow  = na
var float lastJMALow    = na
var float prevJMALow    = na

if not na(pricePH)
    prevPriceHigh := lastPriceHigh
    lastPriceHigh := pricePH
if not na(jmaPH)
    prevJMAHigh := lastJMAHigh
    lastJMAHigh := jmaPH

if not na(pricePL)
    prevPriceLow := lastPriceLow
    lastPriceLow := pricePL
if not na(jmaPL)
    prevJMALow := lastJMALow
    lastJMALow := jmaPL

// Bearish divergence: price HH + JMA LH
bearishDiv = not na(lastPriceHigh) and not na(prevPriceHigh) and
             not na(lastJMAHigh)   and not na(prevJMAHigh)   and
             (lastPriceHigh > prevPriceHigh) and   // Price made higher high
             (lastJMAHigh   < prevJMAHigh)           // JMA made lower high

// Bullish divergence: price LL + JMA HL
bullishDiv = not na(lastPriceLow) and not na(prevPriceLow) and
             not na(lastJMALow)   and not na(prevJMALow)   and
             (lastPriceLow < prevPriceLow) and   // Price made lower low
             (lastJMALow   > prevJMALow)          // JMA made higher low

// Improvement 1: Efficiency gate for divergence
// Divergence is signal only when efficiency is weakening or already weak.
// When VER > 0.55 and VWER > 0.55 (strong trend), divergence = noise.
efficiencyWeakForDiv = ver < ver[1] or vwer < 0.55

// Divergence fires at the pivot confirmation bar — gated by efficiency
bearishDivSignal = bearishDiv and not na(pricePH) and efficiencyWeakForDiv
bullishDivSignal = bullishDiv and not na(pricePL) and efficiencyWeakForDiv

plotshape(showArrows ? bearishDivSignal : na, "Bearish JMA Divergence",
          shape.arrowdown, location.abovebar,
          colDivBear, size=size.small)

plotshape(showArrows ? bullishDivSignal : na, "Bullish JMA Divergence",
          shape.arrowup, location.belowbar,
          colDivBull, size=size.small)

//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
// LAYER 2 — WAE (Volatility Context — preserved, internal use only)
//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

DEAD_ZONE = ta.rma(ta.tr(true), 100) * 3.7

rawVol = switch volType
    "ATR"                => ta.atr(volPeriod)
    "Standard Deviation" => ta.stdev(close, volPeriod)
    "True Range"         => ta.tr
    =>                      ta.atr(volPeriod)

rawVol := na(rawVol) or rawVol == 0 ? 1 : rawVol
volWeight   = rawVol / ta.ema(rawVol, volPeriod)
weightedSrc = close * volWeight

vPhaseR = volJmaPhase < -100 ? 0.5 : volJmaPhase > 100 ? 2.5 : volJmaPhase / 100 + 1.5
vBeta   = 0.45 * (volJmaLen - 1) / (0.45 * (volJmaLen - 1) + 2)
vAlpha  = math.pow(vBeta, volJmaPower)

var float ve0vw = 0.0, ve0vw := (1-vAlpha)*weightedSrc + vAlpha*nz(ve0vw[1],weightedSrc)
var float ve1vw = 0.0, ve1vw := (weightedSrc-ve0vw)*(1-vBeta) + vBeta*nz(ve1vw[1],0)
var float ve2vw = 0.0
var float vjvw  = 0.0
ve2vw := (ve0vw + vPhaseR*ve1vw - vjvw)*math.pow(1-vAlpha,2) + math.pow(vAlpha,2)*nz(ve2vw[1],0)
vjvw  := ve2vw + vjvw

var float ve0v = 0.0,  ve0v  := (1-vAlpha)*volWeight + vAlpha*nz(ve0v[1],volWeight)
var float ve1v = 0.0,  ve1v  := (volWeight-ve0v)*(1-vBeta) + vBeta*nz(ve1v[1],0)
var float ve2v = 0.0
var float vjv  = 0.0
ve2v := (ve0v + vPhaseR*ve1v - vjv)*math.pow(1-vAlpha,2) + math.pow(vAlpha,2)*nz(ve2v[1],0)
vjv  := ve2v + vjv

volJMA = vjv != 0 ? vjvw / vjv : close

isExtHours = useExtHours ? na(time(timeframe.period, "0930-1600", "America/New_York")) : false
avgVol     = ta.sma(volume, vwjmaLen * 2)
isLowVol   = volume < avgVol * (minVolTh / 100.0) or isExtHours
isErratic  = ta.stdev(volume, vwjmaLen) > 0 and
             math.abs(volume - volume[1]) > ta.stdev(volume, vwjmaLen) * 3

useRegJMA  = isLowVol or isErratic
regJMA     = jma_func(close, vwjmaLen, vwjmaPhase, vwjmaPower)

vwPhaseR = vwjmaPhase<-100?0.5:vwjmaPhase>100?2.5:vwjmaPhase/100+1.5
vwBeta   = 0.45*(vwjmaLen-1)/(0.45*(vwjmaLen-1)+2)
vwAlpha  = math.pow(vwBeta, vwjmaPower)

clampVol  = math.max(avgVol*0.1, math.min(volume, avgVol*5))
normVol   = clampVol > 0 ? clampVol : 1.0
vwP       = close * normVol

var float vw0vw=0.0, vw0vw:=(1-vwAlpha)*vwP+vwAlpha*nz(vw0vw[1],vwP)
var float vw1vw=0.0, vw1vw:=(vwP-vw0vw)*(1-vwBeta)+vwBeta*nz(vw1vw[1],0)
var float vw2vw=0.0
var float vwNumer=0.0
vw2vw   := (vw0vw+vwPhaseR*vw1vw-vwNumer)*math.pow(1-vwAlpha,2)+math.pow(vwAlpha,2)*nz(vw2vw[1],0)
vwNumer := vw2vw+vwNumer

var float vw0v=0.0,  vw0v :=(1-vwAlpha)*normVol+vwAlpha*nz(vw0v[1],normVol)
var float vw1v=0.0,  vw1v :=(normVol-vw0v)*(1-vwBeta)+vwBeta*nz(vw1v[1],0)
var float vw2v=0.0
var float vwDenom=0.0
vw2v    := (vw0v+vwPhaseR*vw1v-vwDenom)*math.pow(1-vwAlpha,2)+math.pow(vwAlpha,2)*nz(vw2v[1],0)
vwDenom := vw2v+vwDenom

vwJMA_raw  = vwDenom != 0 ? vwNumer / vwDenom : close
isDeviated = math.abs(vwJMA_raw - close) / close > 0.05
vwJMA_tmp  = (useRegJMA or isDeviated) ? regJMA : vwJMA_raw

sAlpha = 2.0 / (vwjmaSmooth + 1)
var float vwJMA = na
vwJMA := na(vwJMA) ? vwJMA_tmp : sAlpha * vwJMA_tmp + (1 - sAlpha) * vwJMA
if useRegJMA
    eA = 2.0 / (vwjmaSmooth * 2 + 1)
    vwJMA := eA * vwJMA_tmp + (1 - eA) * vwJMA

macdVal  = volJMA - vwJMA
t1       = (macdVal - macdVal[1]) * sensitivity
waeExpansion   = math.abs(t1) > DEAD_ZONE
waeCompression = math.abs(t1) < DEAD_ZONE

//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
// LAYER 3 — LTF CVD (Triangles on Zero Cross)
//
//  Aggregates 30s bar deltas into each 1-min bar.
//  Windowed rolling sum prevents cumulative drift.
//  Triangle fires on zero cross only — not on every rise/fall.
//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

ltf_h = request.security_lower_tf(syminfo.tickerid, ltfTimeframe, high)
ltf_l = request.security_lower_tf(syminfo.tickerid, ltfTimeframe, low)
ltf_c = request.security_lower_tf(syminfo.tickerid, ltfTimeframe, close)
ltf_v = request.security_lower_tf(syminfo.tickerid, ltfTimeframe, volume)

float deltaAgg = 0.0
if array.size(ltf_h) > 0
    for i = 0 to array.size(ltf_h) - 1
        h = array.get(ltf_h, i)
        l = array.get(ltf_l, i)
        c = array.get(ltf_c, i)
        v = array.get(ltf_v, i)
        deltaAgg += (h != l) ? v * (2 * c - h - l) / (h - l) : 0.0

cvd = ta.ema(math.sum(deltaAgg, cvdLen), cvdSmooth)

cvdBullishFlip = ta.crossover(cvd, 0)    // CVD crossed above zero
cvdBearishFlip = ta.crossunder(cvd, 0)   // CVD crossed below zero

// Improvement 2: CVD compression — delta flat while price still moves
// This is the absorptive signal (smart money distributing/accumulating quietly)
// More meaningful than a raw zero-cross because it requires price/delta divergence
atr14           = ta.atr(14)
cvdAvgMove      = ta.sma(math.abs(cvd - cvd[1]), cvdLen)
cvdFlat         = math.abs(cvd - cvd[1]) < (cvdAvgMove * 0.25)   // Delta barely moving
priceActive     = math.abs(close - close[1]) > (atr14 * 0.15)    // But price is moving
cvdCompression  = cvdFlat and priceActive                          // Absorption signal

plotshape(showTriangles ? cvdBullishFlip : na, "CVD Bullish Flip",
          shape.triangleup, location.belowbar,
          colCvdBull, size=size.normal)

plotshape(showTriangles ? cvdBearishFlip : na, "CVD Bearish Flip",
          shape.triangledown, location.abovebar,
          colCvdBear, size=size.normal)

// LAYER 4 — STRUCTURAL BREAKS (Diamonds, Size Scaled)
//
//  Priority hierarchy — highest level wins, no double-counting:
//    Level 3 → Liquidity Sweep  → size.normal  (engineered stop run)
//    Level 2 → Range Rejection  → size.small   (acceptance failure)
//    Level 1 → Micro BOS        → size.tiny    (early structural warning)
//
//  Bullish signals fire below bar (green diamonds)
//  Bearish signals fire above bar (red diamonds)
//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

// Rolling extremes (prior period, no lookahead)
rollHigh = ta.highest(high, sweepLookback)[1]
rollLow  = ta.lowest(low,  sweepLookback)[1]

// Level 3: Liquidity sweep
sweepLow  = low < rollLow  and close > rollLow    // Swept low, closed back inside
sweepHigh = high > rollHigh and close < rollHigh   // Swept high, closed back inside

// Level 2: Range rejection
rngHigh = ta.highest(high, rejectionLen)
rngLow  = ta.lowest(low,  rejectionLen)
rngMid  = (rngHigh + rngLow) / 2

rejectLow  = low  <= rngLow  and close > rngMid   // Tested low, closed above mid
rejectHigh = high >= rngHigh and close < rngMid   // Tested high, closed below mid

// Level 1: Micro BOS (weak close at extreme in trend direction)
mbHigh  = ta.highest(high, microBosLen)
mbLow   = ta.lowest(low,  microBosLen)
barRng  = high - low

weakLow  = barRng > 0 and (close - low)  / barRng < weakCloseTh   // Close near low
weakHigh = barRng > 0 and (high - close) / barRng < weakCloseTh   // Close near high

microBosLong  = slopeNeg and low < mbLow[1]  and weakLow    // New low in downtrend + weak close
microBosShort = slopePos and high > mbHigh[1] and weakHigh  // New high in uptrend + weak close

// Hierarchy: highest level wins
int structBullLevel = sweepLow ? 3 : rejectLow ? 2 : microBosLong ? 1 : 0
int structBearLevel = sweepHigh ? 3 : rejectHigh ? 2 : microBosShort ? 1 : 0

// Improvement 5: Price-based micro BOS decay
// If micro BOS fires but price moves > 0.5 ATR away without a level 2/3 confirming,
// the micro signal is invalidated — stale micro pressure removed from score.
var int   microBullBar   = na
var float microBullClose = na
var int   microBearBar   = na
var float microBearClose = na

// Record bar/close when micro fires (level 1 only)
if structBullLevel == 1
    microBullBar   := bar_index
    microBullClose := close
if structBullLevel >= 2   // Level 2/3 confirmed — reset micro tracker
    microBullBar   := na
    microBullClose := na

if structBearLevel == 1
    microBearBar   := bar_index
    microBearClose := close
if structBearLevel >= 2
    microBearBar   := na
    microBearClose := na

// Decay condition: micro fired, no level 2/3 since, price moved > 0.5 ATR
microBullDecayed = not na(microBullBar) and structBullLevel < 2 and
                   math.abs(close - microBullClose) > atr14 * 0.5

microBearDecayed = not na(microBearBar) and structBearLevel < 2 and
                   math.abs(close - microBearClose) > atr14 * 0.5

// Effective levels after decay applied
effectiveBullLevel = microBullDecayed ? 0 : structBullLevel
effectiveBearLevel = microBearDecayed ? 0 : structBearLevel

// Bullish diamonds (below bar) — one plotshape per level, gated by level match
plotshape(showDiamonds ? (structBullLevel == 3) : na, "Sweep Low",
          shape.diamond, location.belowbar, colSweepBull, size=size.normal)

plotshape(showDiamonds ? (structBullLevel == 2) : na, "Reject Low",
          shape.diamond, location.belowbar, colRejectBull, size=size.small)

plotshape(showDiamonds ? (structBullLevel == 1) : na, "Micro BOS Long",
          shape.diamond, location.belowbar, colMicoBull, size=size.tiny)

// Bearish diamonds (above bar) — one plotshape per level
plotshape(showDiamonds ? (structBearLevel == 3) : na, "Sweep High",
          shape.diamond, location.abovebar, colSweepBear, size=size.normal)

plotshape(showDiamonds ? (structBearLevel == 2) : na, "Reject High",
          shape.diamond, location.abovebar, colRejectBear, size=size.small)

plotshape(showDiamonds ? (structBearLevel == 1) : na, "Micro BOS Short",
          shape.diamond, location.abovebar, colMicoBear, size=size.tiny)

// LAYER 5 — VER + VWER (computed in Layer 0b above — see score section)
//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
// LAYER 6 — HTF VER + VWER (Market State Histogram)
//
//  Runs VER and VWER on 5-minute bars to classify the broader session regime.
//  VER (volatility-weighted ER) — penalises choppy path
//  VWER (volume-weighted ER)    — penalises low participation
//
//  Four histogram states (color changes on regime shift):
//    GREEN  — Both > htfTrendTh      → Trending session, momentum works
//    YELLOW — VER falling + VWER OK  → Deteriorating, regime shifting
//    RED    — Either < htfReversalTh → Ranging/exhausted, fade extremes
//    GRAY   — Neutral / transitional
//
//  Bar height = htfVER value (0 to ~1+)
//  The histogram sits in the same pane and auto-scales with the score line.
//  Both series share the 0-1 range so they coexist cleanly.
//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

[htfVER, htfVWER] = request.security(syminfo.tickerid, htfTF,
     [ta.ema(
         math.abs(close - close[htfEffLen]) /
         math.max(
             (math.sum(math.abs(close - close[1]), htfEffLen)) == 0 ? 1.0 :
              math.sum(math.abs(close - close[1]), htfEffLen), 1.0) /
         math.max(ta.sma(math.abs(close - ta.sma(close, htfPathLen)), htfPathLen), 1.0),
         htfVerSmooth),

         math.abs(close - close[htfEffLen]) /
         math.max(

              math.sum(math.abs(close - close[1]), htfEffLen), 1.0) /
         math.max(ta.sma(volume, htfVolLen) != 0 ? volume / ta.sma(volume, htfVolLen) : 1.0, 0.01),
         htfVerSmooth)],
     lookahead=barmerge.lookahead_off)

htfRanging     = htfVER < htfReversalTh or htfVWER < htfReversalTh
