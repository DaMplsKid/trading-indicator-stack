jma_func(_src, _length, _phase, _power) =>
    phaseRatio = _phase < -100 ? 0.5 : _phase > 100 ? 2.5 : _phase / 100 + 1.5
    beta  = 0.45 * (_length - 1) / (0.45 * (_length - 1) + 2)
    alpha = math.pow(beta, _power)
    var float e0 = na
    var float e1 = na
    var float e2 = na
    var float jv = na
    e0 := (1 - alpha) * _src + alpha * nz(e0[1], _src)
    e1 := (_src - e0) * (1 - beta) + beta * nz(e1[1], 0)
    e2 := (e0 + phaseRatio * e1 - nz(jv[1], _src)) * math.pow(1 - alpha, 2) + math.pow(alpha, 2) * nz(e2[1], 0)
    jv  := e2 + nz(jv[1], _src)
    jv

// HTF1 SECURITY — 1hr KALMAN + MAD BANDS
//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

htf1_block() =>
    var float kalman_est = na
    var float kalman_p   = 1.0
    
    if na(kalman_est)
        kalman_est := close
    
    pred_est = kalman_est
    pred_p   = kalman_p + kalman_q
    
    K = pred_p / (pred_p + kalman_r)
    kalman_est := pred_est + K * (close - pred_est)
    kalman_p   := (1 - K) * pred_p
    
    median_val = ta.median(close, mad_len)
    float mad_sum = 0.0
    for i = 0 to mad_len - 1
        mad_sum += math.abs(close[i] - median_val)
    mad = mad_sum / mad_len
    
    upper = kalman_est + mad * mad_mult
    lower = kalman_est - mad * mad_mult
    
    slope = kalman_est - nz(kalman_est[1])
    
    [kalman_est, upper, lower, slope]

// HTF2 SECURITY — 5min SUPER SMOOTHER + PRESSURE BANDS
//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

htf2_block() =>
    a1 = math.exp(-math.sqrt(2.0) * math.pi / ss_period)
    b1 = 2.0 * a1 * math.cos(math.sqrt(2.0) * math.pi / ss_period)
    c2 = b1
    c3 = -(a1 * a1)
    c1 = 1.0 - c2 - c3
    
    var float ss = na
    ss := na(ss[1]) or na(ss[2]) ? close :
          c1 * (close + nz(close[1])) * 0.5 + c2 * nz(ss[1]) + c3 * nz(ss[2])
    
    ss_slope = ss - nz(ss[1])
    
    atr_val = ta.atr(14)
    net = math.abs(ss_slope - ss_slope[pressure_len])
    float tot = 0.0
    for i = 0 to pressure_len - 1
        tot += math.abs(ss_slope[i] - nz(ss_slope[i + 1]))
    eff = tot != 0 ? net / tot : 0.0
    pressure = ta.ema((ss_slope - nz(ss_slope[1])) * eff, pressure_len)
    norm_pressure = atr_val != 0 ? pressure / atr_val : 0.0
    
    press_raw = norm_pressure * atr_val
    press_std = ta.stdev(press_raw, pressure_len) * pressure_mult
    
    upper = ss + press_std
    lower = ss - press_std
    
    [ss, upper, lower, norm_pressure, ss_slope]

iTrend_func(src, alpha) =>
    var float it = na
    it := (alpha - (alpha*alpha)/4) * src + 
          0.5 * alpha * alpha * nz(src[1]) - 
          (alpha - 0.75*alpha*alpha) * nz(src[2]) + 
          2 * (1 - alpha) * nz(it[1]) - 
          (1 - alpha) * (1 - alpha) * nz(it[2])
    it

// MICROSTRUCTURE INDICATORS (NEW)
//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

float vol_surge_score = 0.0
float tape_speed_score = 0.0
float micro_score = 50.0  // Default neutral score

if enable_micro
    // Volume Acceleration - detects surge without L2
    vol_avg = ta.sma(volume, vol_accel_len)
    vol_surge = volume > (vol_avg * vol_surge_mult)
    
    // Volume quality (surge + directional)
    vol_surge_bullish = vol_surge and close > open
    vol_surge_bearish = vol_surge and close < open
    
    vol_surge_score := vol_surge_bullish ? 100.0 : vol_surge_bearish ? 0.0 : 50.0
    
    // Tape Speed - measures bars/minute acceleration
    // Higher speed = more activity = potential breakout
    var int bar_count = 0
    var int last_minute = na
    
    current_minute = minute(time)
    if na(last_minute)
        last_minute := current_minute
    
    if current_minute != last_minute
        last_minute := current_minute
        bar_count := 1
    else
        bar_count += 1
    
    // Historical bars per minute
    var float bars_per_min_avg = 1.0
    bars_per_min_avg := ta.ema(bar_count, tape_speed_len)
    
    // Acceleration ratio
    tape_accel = bars_per_min_avg > 0 ? bar_count / bars_per_min_avg : 1.0
    
    // Score tape speed
    tape_speed_score := tape_accel > 1.3 ? 100.0 : tape_accel > 1.1 ? 75.0 : tape_accel > 0.9 ? 50.0 : 25.0 
    
    // Combined microstructure score
    micro_score := (vol_surge_score * 0.6) + (tape_speed_score * 0.4)

//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
// SETUP QUALITY GRADING SYSTEM (SCALPER EDITION)
//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

float setup_score = na
string setup_grade = na

if enable_grading
    float total_weight = weight_htf1 + weight_htf2 + weight_ctf + weight_price + weight_micro
    float w1 = weight_htf1 / total_weight
    float w2 = weight_htf2 / total_weight
    float w3 = weight_ctf / total_weight
    float w4 = weight_price / total_weight
    float w5 = weight_micro / total_weight
    
    // Factor 1: HTF1 Alignment
    jma_bullish = jma_slope_raw > 0
    jma_bearish = jma_slope_raw < 0
    
    htf1_aligned = (htf1_bias_up and jma_bullish) or (htf1_bias_dn and jma_bearish)
    htf1_score = htf1_aligned ? 100.0 : 0.0
    
    // Factor 2: HTF2 Alignment
    htf2_bullish = htf2_slope > 0
    htf2_bearish = htf2_slope < 0
    
    htf2_aligned = htf2_trending and ((htf2_bullish and jma_bullish) or (htf2_bearish and jma_bearish))
    htf2_partial = htf2_chop
    
    htf2_score = htf2_aligned ? 100.0 : htf2_partial ? 50.0 : 0.0
    
    // Factor 3: CTF Quality
    ctf_score = jma_green ? 100.0 : jma_yellow ? 70.0 : jma_blue ? 30.0 : 0.0
    
    // Factor 4: Price Position
    price_score = 100.0
    
    if htf2_overbought
        price_score := 30.0
    else if htf2_oversold
        price_score := 70.0
    
    if outside_upper
        price_score := math.min(price_score, 50.0)
    
    // Factor 5: Microstructure (NEW)
    micro_score_final = micro_score
    
    // Calculate weighted setup score
    setup_score := (htf1_score * w1) + (htf2_score * w2) + (ctf_score * w3) + (price_score * w4) + (micro_score_final * w5)
    setup_score := math.max(0.0, math.min(100.0, setup_score))
    
    // Assign grade
    setup_grade := setup_score >= elite_threshold ? "A+" :
                   setup_score >= strong_threshold ? "A" :
                   setup_score >= moderate_threshold ? "B" :
                   "C"

// LONG ONLY GATE - Only show setups when JMA bullish
jma_bullish_confirmed = jma_slope_raw > 0
